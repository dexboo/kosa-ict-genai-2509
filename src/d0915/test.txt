# 환경 설정


1. wsl2 설치
제어판 > 프로그램 > 프로그램 및 기능 > Windows 기능 켜기 또는 끄기
재부팅
wsl --install ubuntu-22.04		# 우분투 설치 후 계정 생성
wsl -l -v		# 배포판 확인
wsl -d ubuntu-22.04		# 콘솔 진입
wsl --terminate ubuntu-22.04		# 종료
wsl --unregister ubuntu-22.04		# 배포판 삭제

2. git 설치
mkdir work
cd work
git config --global user.name '깃이름'
git config --global user.email '깃이메일'
git config --global --list

3. ssh키 설정
ssh-keygen		# ssh 키 생성
cat ~/.ssh/id_rsa.pub		# ssh 퍼블릭키 복사
git > Settings > SSH and GPG keys > New SSH key > 키 등록

4. git 복사 및 연습 pull & push
git clone git@github.com:weable-kosa/kosa-ict-genai-2025-2nd.git
git pull
git push
